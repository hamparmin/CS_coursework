stan_code = '''

    int<lower=1> n;    // len of data
    vector[n] y;  // data
    real<lower=0> alpha;  // prior hyperparameter
    real<lower=0> beta;   // prior hyperparameter
}

parameters {
    real<lower=0> lambda;
}

model {
    lambda ~ gamma(alpha,beta); // gamma prior
    y ~ exponential(lambda);  // likelihood function
}
'''
#compile model
stan_model = pystan.StanModel(model_code=stan_code)

stan_code = """
data {
    int<lower=0> n; // len of data
    real y[n]; // data
    real mu_0; //prior hyperparameters
    real nu_0;
    real alpha_0;
    real beta_0;
}

parameters {
    real mu; //data mean
    real<lower=0> sigma; //data variance
}

model {
    sigma ~ inv_gamma(alpha_0, beta_0); // variance
    mu ~ normal(mu_0, sqrt(sigma/ nu_0)); // mean    
    y ~ normal(mu, sqrt(sigma)); // likelihood
}
"""
stan_model = pystan.StanModel(model_code=stan_code)
